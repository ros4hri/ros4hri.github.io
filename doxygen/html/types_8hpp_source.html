240enum class FeatureType
241{
242 kInvalid = 0,
243 kPerson = (1u << 0), // all known persons, whether or not they are currently seen
244 kTrackedPerson = (1u << 1), // only the actively tracked persons
245 kFace = (1u << 2),
246 kBody = (1u << 3),
247 kVoice = (1u << 4)
248}; // note that FeatureType values can also be used as bitmasks
249
250inline FeatureType operator&(const FeatureType & lhs, const FeatureType & rhs)
251{
252 return static_cast<FeatureType>(
253 std::underlying_type<FeatureType>::type(lhs) & std::underlying_type<FeatureType>::type(rhs));
254}
255
256inline FeatureType operator|(const FeatureType & lhs, const FeatureType & rhs)
257{
258 return static_cast<FeatureType>(
259 std::underlying_type<FeatureType>::type(lhs) | std::underlying_type<FeatureType>::type(rhs));
260}
